# Optimised: ~6× faster in CPython
def _xor_many(values):
    if not values: return b""
    n = len(values[0])
    acc = int.from_bytes(values[0], "big")
    for v in values[1:]:
        if len(v) != n:
            raise ValueError("length mismatch")
        acc ^= int.from_bytes(v, "big")
    return acc.to_bytes(n, "big")

hash.py · hash_lms() — domain-separated SHA-256
def hash_lms(tweak, *data):
    h = hashlib.sha256()
    for t in tweak:          # encode each tweak element as 4-byte big-endian
        if isinstance(t, int):
            h.update(t.to_bytes(4, "big"))
        else:
            h.update(t)
    for d in data:
        h.update(d)
    return h.digest()
distributed_signing.py · KK_SetupContribution() — trustee Round 1 + Round 2 seed
def KK_SetupContribution(seed, KeyID, R, path_lens, sk_shape):
    rows, cols = sk_shape
    return {
        "Rt":    PRF_R(seed, KeyID, N_BYTES),       # tweak (30, keyId)
        "CHKt":  PRF_Chk(seed, KeyID, N_BYTES),     # tweak (31, keyId)
        "Auth":  PRF_Auth(seed, KeyID, R),          # tweak (32, keyId)
        "PATHt": [PRF_Path(seed, KeyID, idx, n)
                  for idx, n in enumerate(path_lens)],
        "SKt":   [[PRF_Chain(seed, KeyID, i, j, N_BYTES)
                   for j in range(cols)]
                  for i in range(rows)],
    }
